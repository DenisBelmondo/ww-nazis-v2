ACTOR HNaziBerserk : CustomInventory replaces ArtiTomeOfPower
{
	Game Heretic
	Inventory.PickupMessage "Foliant der Macht" // "Tome of Power"
	Inventory.PickupSound "misc/p_pkup"
	Tag "Foliant der Macht"
	+COUNTITEM
	+FLOATBOB
	+INVENTORY.PICKUPFLASH
	+INVENTORY.INVBAR
	Inventory.Icon "ARTIPWBK"
	Inventory.MaxAmount 25
	States
	{
	Spawn:
		PWBK A 350
		Loop
	Use:
		TNT1 A 0 A_GiveInventory("PowerStrength")
		TNT1 A 0 HealThing(100,0)
		Stop
	}
}

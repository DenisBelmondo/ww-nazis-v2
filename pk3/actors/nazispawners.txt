//These allow for choosing between Nazis and IWAD default monsters, primarily inspired
//by how Heretic and Chex Quest support were set up. However, this can impact maps
//with heavy scripting for monster spawning and activation, so to nix this, 'unlock'
//the MonsterToggle's inheritence from RandomSpawner to use the DropItem properties.
ACTOR MonsterToggle //: RandomSpawner
{
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("MonsterType")==1,"Nazis")
		TNT1 A 0 A_Jump(256,"Default")
		Stop
	}
}

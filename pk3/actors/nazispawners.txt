//These allow for choosing between Nazis and IWAD default monsters, primarily inspired
//by how Heretic and Chex Quest support were set up.
ACTOR DormantToggle : CustomInventory
{
	+INVENTORY.ALWAYSPICKUP
	Inventory.MaxAmount 0
	States
	{
	Pickup:
		TNT1 A 0 Thing_Deactivate(0)
		Stop
	}
}
ACTOR DormantToggleX : DormantToggle
{
	States
	{
	Pickup:
		TNT1 A 0 A_GiveToChildren("DormantToggle")
		Stop
	}
}
ACTOR MonsterToggle : SwitchableDecoration
{
	+NOCLIP
	+SHOOTABLE
	+NONSHOOTABLE
	States
	{
	Spawn:
	Inactive:
		TNT1 A 0 ACS_NamedExecuteAlways("SleepChildren",0)
	Active:
		TNT1 A 0 A_JumpIf(CallACS("MonsterType")==1,"Nazis")
		TNT1 A 0 A_Jump(256,"Default")
		Stop
	Death:
		TNT1 A 1 A_KillMaster
		Stop
	}
}

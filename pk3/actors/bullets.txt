ACTOR Tracer : FastProjectile
{
	Alpha 0.90
	RenderStyle Add
	Radius 2
	Height 2
	Speed 75
	PROJECTILE
	+BLOODSPLATTER
	Scale 0.4
	MissileType TracerTrail
	MissileHeight 8
	Decal BulletChip
	States
	{
	Spawn:
		PUFF A -1 BRIGHT
		Stop
	Death:
	Crash:
		PUFF A 0 A_SpawnDebris("TracerSpark")
		PUFF A 2 BRIGHT A_PlaySound("ricochet")
		PUFF BCD 2 BRIGHT
		Stop
	XDeath:
		TNT1 A 1 A_PlaySound("hitflesh")
		Stop
	}
}

	-NOBLOCKMAP

	BounceSound "bulletcasing/bounce"
	BounceFactor 0.5
	WallBounceFactor 0.2

	Height 3
